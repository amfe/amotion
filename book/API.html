@class lib.motion~Motion
@param {Object} config 形如{v, a, s, t}，其中t和s最少知其一，即可推导出另一个
@param {Number} config.v 已知初速度
@param {Number} config.a 已知加速度
@param {Number} config.s 位移
@param {Number} config.t 时间
